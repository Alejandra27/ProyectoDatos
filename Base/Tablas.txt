
  
  create table VUELO(
  ID NUMBER(20) NOT NULL,
  PASAJERO VARCHAR(20) NOT NULL,
  ORIGEN VARCHAR(20) NOT NULL,
  DESTINO VARCHAR(20) NOT NULL,
  CONSTRAINT PK_VUELO PRIMARY KEY(ID),
  CONSTRAINT PK_VUELO_PASAJEROS FOREIGN KEY (PASAJERO) REFERENCES PASAJEROS(DOCUMENTO),
  CONSTRAINT PK_VUELO_ORIGEN FOREIGN KEY (ORIGEN) REFERENCES CUIDAD(CODIGO),
  CONSTRAINT PK_VUELO_DESTINO FOREIGN KEY (DESTINO) REFERENCES CIUDAD(CODIGO)
);
  
  create table CIUDAD(
  CODIGO NUMBER(20) NOT NULL,
  NOMBRE VARCHAR(80) NOT NULL,
  CONSTRAINT PK_CUIDAD PRIMARY KEY (CODIGO)
);
  
  
  
  
  create table PASAJEROS (nombre varChar(20), cedula varChar(20), silla varChar(20), genero varChar(20), edad varChar(20));
  
  create table AEROLINEA (ciudades varChar(20), precio varChar(20));
